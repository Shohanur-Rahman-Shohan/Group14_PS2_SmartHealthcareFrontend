Train Disease Prediction Model

# Load data from Patients and MedicalRecords tables
patients = db.query("SELECT * FROM Patients WHERE age > 18")
records = db.query("""
  SELECT mr.diagnosis, mr.treatment, p.age, p.gender
  FROM MedicalRecords mr
  JOIN Patients p ON mr.PatientID = p.PatientID
""")

# Prepare features for ML model
features = preprocess_data(patients, records)
labels = extract_diagnosis_labels(records)

# Train model
model = DiseasePredictor()
model.fit(features, labels)

# Save predictions to DiseasePredictions table
predictions = model.predict(new_symptoms)
db.insert("DiseasePredictions", predictions)

Generate Health Recommendations

-- Query to generate personalized health recommendations
INSERT INTO Recommendations (PatientID, recommendation, priority)
SELECT 
  p.PatientID,
  CASE 
    WHEN mr.diagnosis = 'Diabetes' THEN 
      'Schedule quarterly A1C test and eye examination'
    WHEN p.age > 50 THEN 
      'Consider annual preventive health screening'
    ELSE 'Maintain regular exercise and balanced diet'
  END as recommendation,
  'Medium' as priority
FROM Patients p
LEFT JOIN MedicalRecords mr ON p.PatientID = mr.PatientID
WHERE p.last_checkup < DATE_SUB(NOW(), INTERVAL 1 YEAR)
OR mr.diagnosis IN ('Diabetes', 'Hypertension', 'Heart Disease');
